CREATE DATABASE projetocarrinho;
USE projetocarrinho;

CREATE TABLE estoque (
    id INT PRIMARY KEY,
    nome VARCHAR(50),
    categoria VARCHAR(50),
    valor DECIMAL(10, 2),
    quantidade INT
);

CREATE TABLE carrinho (
    id INT PRIMARY KEY,
    produto_id INT,
    quantidade INT,
    valor_total DECIMAL(10, 2),
    FOREIGN KEY (produto_id) REFERENCES estoque(id)
);
